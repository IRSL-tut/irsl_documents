### install sphinx

(apt update)
sudo apt install pandoc

git clone https://github.com/IRSL-tut/irsl_documents.git
cd irsl_documents

cd util

python3 -m pip install virtualenv
virtualenv --python=python3 sphinx

. sphinx/bin/activate

python3 -m pip install -r requirements.txt
